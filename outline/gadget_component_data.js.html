/**
 * @fileoverview
 * A gadget component data object holds information on a gadget component.
 * This data is used to instantiate gadget components. A gadget component must
 * be instantiated from a catalog component data object. In addition to holding
 * core component data like longname, price; the object also holds additional
 * metadata such as a name or geometry.
 */

goog.provide('gadgetron.gadget.ComponentData');


/**
 * This object holds information about a gadget component. A gadget component
 * must be instantiated from a catalog component data.
 * @param {gadgetron.common.AppContext} appContext Application context.
 * @param {!gadgetron.catalog.ComponentData} catalogComponentData Catalog data.
 * @constructor
 */
gadgetron.gadget.ComponentData = function(appContext, catalogComponentData) {
  /**
   * Long name.
   * @type {!string}
   * @private
   */
  this.longname_ = wrapped.getLongname();

  /**
   * Key name.
   * @type {!string}
   * @private
   */
  this.keyname_ = wrapped.getKeyname();

  /**
   * Price.
   * @type {!number}
   * @private
   */
  this.price_ = wrapped.getPrice();

  /**
   * Component URL.
   * @type {!string}
   * @private
   */
  this.componentUrl_ = wrapped.getComponentUrl();

  /**
   * SVG URL.
   * @type {!string}
   * @private
   */
  this.svgUrl_ = wrapped.getSvgUrl();

  /** METADATA */

  /**
   * A name provided by the user.
   * @type {string}
   * @private
   */
  this.name_ = this.longname_;

  /**
   * Transformation data.
   * @type {gadgetron.common.TransformationData}
   * @private
   */
  this.transformation_ = null;
};
