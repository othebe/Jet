/**
 * @fileoverview
 * This wraps catalog component data read in as JSON.
 */

goog.provide('gadgetron.catalog.ComponentJson');

goog.require('gadgetron.catalog.IComponent');


/**
 * Component wrapper for JSON data.
 * @param {gadgetron.common.AppContext} appContext Application context.
 * @param {{
 * 	longname: string,
 * 	keyname: string,
 * 	price: number,
 * 	componentUrl: string,
 * 	svgUrl: string
 * }} componentJson Component JSON.
 * @implements {gadgetron.catalog.IComponent}
 * @constructor
 */
gadgetron.common.wrapper.ComponentJson = function(appContext, componentJson) {
  // Check integrity of JSON object.
  this.checkAssertions_(componentJson);

  /**
   * Long name.
   * @type {!string}
   * @private
   */
  this.longname_ = wrapped.getLongname();

  /**
   * Key name.
   * @type {!string}
   * @private
   */
  this.keyname_ = wrapped.getKeyname();

  /**
   * Price.
   * @type {!number}
   * @private
   */
  this.price_ = wrapped.getPrice();

  /**
   * Component URL.
   * @type {!string}
   * @private
   */
  this.componentUrl_ = wrapped.getComponentUrl();

  /**
   * SVG URL.
   * @type {!string}
   * @private
   */
  this.svgUrl_ = wrapped.getSvgUrl();
};


/**
 * Check that the JSON object contains required fields.
 * @param {Object} obj The JSON object.
 * @private
 */
gadgetron.common.wrapper.ComponentJson.prototype.checkAssertions_ =
    function(obj) {
   goog.asserts.assertString(obj.longname);
   goog.asserts.assertString(obj.keyname);
   goog.asserts.assertNumber(obj.price);
   goog.asserts.assertString(obj.componentUrl);
   goog.asserts.assertString(obj.svgUrl);
};


/** @inheritDoc */
gadgetron.common.wrapper.ComponentJson.prototype.getLongname = function() {
  return this.longname_;
};


/** @inheritDoc */
gadgetron.common.wrapper.ComponentJson.prototype.getKeyname = function() {
  return this.keyname_;
};


/** @inheritDoc */
gadgetron.common.wrapper.ComponentJson.prototype.getPrice = function() {
  return this.price_;
};


/** @inheritDoc */
gadgetron.common.wrapper.ComponentJson.prototype.getComponentUrl = function() {
  return this.componentUrl_;
};


/** @inheritDoc */
ComponentJson.prototype.getSvgUrl = function() {
  return this.svgUrl_;
};
