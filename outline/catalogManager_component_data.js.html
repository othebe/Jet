/**
 * @fileoverview
 * A catalog component data object holds information on a catalog component.
 * This data is used to instantiate catalog components.
 */

goog.provide('gadgetron.catalogManager.ComponentData');


/**
 * This object holds information about a catalog component.
 * @param {gadgetron.common.AppContext} appContext Application context.
 * @param {!gadgetron.componentManager.IComponent} wrapped Wrapped component.
 * @constructor
 */
gadgetron.catalogManager.ComponentData = function(appContext, wrapped) {
  /**
   * Long name.
   * @type {!string}
   * @private
   */
  this.longname_ = wrapped.getLongname();

  /**
   * Key name.
   * @type {!string}
   * @private
   */
  this.keyname_ = wrapped.getKeyname();

  /**
   * Price.
   * @type {!number}
   * @private
   */
  this.price_ = wrapped.getPrice();

  /**
   * Component URL.
   * @type {!string}
   * @private
   */
  this.componentUrl_ = wrapped.getComponentUrl();

  /**
   * SVG URL.
   * @type {!string}
   * @private
   */
  this.svgUrl_ = wrapped.getSvgUrl();
};


/** @return {string} */
gadgetron.catalogManager.ComponentData.prototype.getLongname = function() {
  return this.longname_;
};


/** @return {string} */
gadgetron.catalogManager.ComponentData.prototype.getKeyname = function() {
  return this.keyname_;
};


/** @return {number} */
gadgetron.catalogManager.ComponentData.prototype.getPrice = function() {
  return this.price_;
};


/** @return {string} */
gadgetron.catalogManager.ComponentData.prototype.getComponentUrl = function() {
  return this.componentUrl_;
};


/** @return {string} */
gadgetron.catalogManager.ComponentData.prototype.getSvgUrl = function() {
  return this.svgUrl_;
};
